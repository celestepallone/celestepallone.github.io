  <script>

    function toggleDetail(group, n) {

      const allCards = document.querySelectorAll('#' + (group === 'research' ? 'research' : 'photos') + ' .clickable-card');

      const allDetails = document.querySelectorAll('[id^="' + group + '-detail-"]');

      const detail = document.getElementById(group + '-detail-' + n);

      const card = allCards[n - 1];

      const isOpen = detail.classList.contains('visible');


      allDetails.forEach(d => d.classList.remove('visible'));

      allCards.forEach(c => c.classList.remove('active'));


      if (!isOpen) {

        detail.classList.add('visible');

        card.classList.add('active');

      }

    }

  </script>
